    private void jButton3ActionPerformed(java.awt.event.ActionEvent evt) {
        // TODO add your handling code here:
        if ((a1.isSelected() || b1.isSelected() || c1.isSelected() || d1.isSelected())&&(a2.isSelected() || b2.isSelected() || c2.isSelected() || d2.isSelected())
            &&(a3.isSelected() || b3.isSelected() || c3.isSelected() || d3.isSelected())&&(a4.isSelected() || b4.isSelected() || c4.isSelected() || d4.isSelected())){

            if (a1.isSelected()&& d2.isSelected() && a3.isSelected() && d4.isSelected()){
                mp = 4;}
            else if (a1.isSelected()&& d2.isSelected() && a3.isSelected()){
                mp= 3;}
            else if (a1.isSelected()&& d2.isSelected()){
                mp =2;}
            else if(a1.isSelected()){
                mp=1;}
            else {mp = 0;}
            System.out.print(""+mp);

            int ans = JOptionPane.showConfirmDialog(null,"Scores submitted! Wanna check your ans.?");
            if (ans== JOptionPane.YES_OPTION){
                if (a1.isSelected()&& d2.isSelected() && a3.isSelected() && d4.isSelected()){
                    JOptionPane.showMessageDialog(null,"Ques1 correct,  Ques2 correct,   Ques3 correct,   Ques4 correct");}

                else if (a1.isSelected()&& d2.isSelected() && a3.isSelected()){
                    JOptionPane.showMessageDialog(null,"Ques1 correct,   Ques2 correct,   Ques3 correct,  Ques4 incorrect(sol.D)");}
                else if (a1.isSelected()&& d2.isSelected()){
                    JOptionPane.showMessageDialog(null,"Ques1 correct,   Ques2 correct,   Ques3 incorrect(sol.A),   Ques4 incorrect(sol.D)");}
                else if(a1.isSelected()){
                    JOptionPane.showMessageDialog(null,"Ques1 correct,   Ques2 incorrect(sol.D),   Ques3 incorrect(sol.A),   Ques4 incorrect(sol.D)");}
                else {JOptionPane.showMessageDialog(null,"Ques1 incorrect(sol.A),   Ques2 incorrect(sol.D),   Ques3 incorrect(sol.A),   Ques4 incorrect(sol.D)");}

                new set1sub().setVisible(true);
                this.setVisible(false);
            }

            else if(ans== JOptionPane.NO_OPTION)
            {new set1sub().setVisible(true);
                this.setVisible(false);}}
        else {JOptionPane.showMessageDialog(null,"Please answer all the Questions!");}

    }
